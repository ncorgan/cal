INCLUDE_DIRECTORIES(
    ${CMAKE_SOURCE_DIR}/include
    ${CMAKE_CURRENT_SOURCE_DIR}
)

SET(cal_srcs
    net/ifaddrs.c
)
IF(WIN32)
    SET(cal_libs
		iphlpapi
        ntdll
        shlwapi
        ws2_32
    )
ELSE()
    SET(cal_libs
        dl
    )
ENDIF()

ADD_LIBRARY(cal STATIC ${cal_srcs})
TARGET_LINK_LIBRARIES(cal ${cal_libs})
